import pandas as pd
import numpy as np
from sklearn.decomposition import KernelPCA
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score
import matplotlib.pyplot as plt

# Load the design parameter data
input_data = pd.read_csv('hw1_input.csv')

# Load the real and imaginary parts of S11
real_data = pd.read_csv('hw1_real.csv')
imag_data = pd.read_csv('hw1_img.csv')

# Calculate the magnitude of S11
magnitude = np.sqrt(real_data**2 + imag_data**2)

# Find the frequency index where the minimum magnitude occurs for each row
selected_frequencies = magnitude.idxmin(axis=1).astype(int)  # Ensure indices are integers

# Extract the real and imaginary parts at the selected frequencies
y_real = real_data.to_numpy()[np.arange(len(real_data)), selected_frequencies]
y_imag = imag_data.to_numpy()[np.arange(len(imag_data)), selected_frequencies]

# Apply Kernel PCA to the input data for dimensionality reduction
n_components = 4  # Choose the desired number of components
kernel_pca = KernelPCA(n_components=n_components, kernel='rbf', gamma=0.1)  # Adjust gamma based on data
X_kpca = kernel_pca.fit_transform(input_data)

# Initialize and fit linear regression models for the Kernel PCA transformed data
model_real = LinearRegression()
model_imag = LinearRegression()

model_real.fit(X_kpca, y_real)
model_imag.fit(X_kpca, y_imag)

# Predict the real and imaginary components of S11 at selected frequency points
y_real_pred = model_real.predict(X_kpca)
y_imag_pred = model_imag.predict(X_kpca)

# Evaluate the models using Mean Squared Error and R^2 score
mse_real = mean_squared_error(y_real, y_real_pred)
r2_real = r2_score(y_real, y_real_pred)
mse_imag = mean_squared_error(y_imag, y_imag_pred)
r2_imag = r2_score(y_imag, y_imag_pred)

print(f"Real Part - Mean Squared Error: {mse_real:.4f}, R^2 Score: {r2_real:.4f}")
print(f"Imaginary Part - Mean Squared Error: {mse_imag:.4f}, R^2 Score: {r2_imag:.4f}")

# Plot actual vs predicted values for the real part at selected frequency points
plt.figure(figsize=(12, 6))
plt.scatter(y_real, y_real_pred, edgecolor='k', alpha=0.7)
plt.plot([y_real.min(), y_real.max()], [y_real.min(), y_real.max()], 'r--', label="Ideal Fit")
plt.xlabel('Actual Real Part')
plt.ylabel('Predicted Real Part')
plt.title('Real Part at Selected Frequency Points - Actual vs Predicted (Kernel PCA)')
plt.legend()
plt.grid(True)
plt.show()

# Plot actual vs predicted values for the imaginary part at selected frequency points
plt.figure(figsize=(12, 6))
plt.scatter(y_imag, y_imag_pred, edgecolor='k', alpha=0.7)
plt.plot([y_imag.min(), y_imag.max()], [y_imag.min(), y_imag.max()], 'r--', label="Ideal Fit")
plt.xlabel('Actual Imaginary Part')
plt.ylabel('Predicted Imaginary Part')
plt.title('Imaginary Part at Selected Frequency Points - Actual vs Predicted (Kernel PCA)')
plt.legend()
plt.grid(True)
